56  virtual void update_audio_data(Frame* data, uint16_t frameCount) {
57  if (data!=nullptr && frameCount>0 && ( mono_downmix || is_volume_used)) {
58  ESP_LOGD("VolumeControl", "update_audio_data");
59  for (int i=0;i<frameCount;i++){
60  int32_t pcmLeft = data[i].channel1;
61  int32_t pcmRight = data[i].channel2;
62  // if mono -> we provide the same output on both channels
63  if (mono_downmix) {
64  pcmRight = pcmLeft = (pcmLeft + pcmRight) / 2;
65  }
66  // adjust the volume
67  if (is_volume_used) {
68  pcmLeft = pcmLeft * volumeFactor / volumeFactorMax;
69  pcmRight = pcmRight * volumeFactor / volumeFactorMax;
70  }
71  data[i].channel1 = pcmLeft;
72  data[i].channel2 = pcmRight;
73  }
74  }
75  }
76 
77  // provides a factor in the range of 0 to 4096

110 class A2DPDefaultVolumeControl : public A2DPVolumeControl {
111  protected:
112  void set_volume(uint8_t volume) override {
113  constexpr float base = 1.4f;
114  constexpr float bits = 12.0f;
115  constexpr float zero_ofs = pow(base, -bits);
116  constexpr float scale = pow(2.0f, bits);
117  float volumeFactorFloat = (pow(base, volume * bits / 127.0f - bits) - zero_ofs) * scale / (1.0f - zero_ofs);
118  volumeFactor = volumeFactorFloat;
119  if (volumeFactor > 0x1000) {
120  volumeFactor = 0x1000;
121  }
122  }
123 };
