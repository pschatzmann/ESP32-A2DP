27 class VolumeControl {
28  public:
29  virtual void update_audio_data(Frame* data, uint16_t frameCount, uint8_t volume, bool mono_downmix, bool is_volume_used) {
30  if (mono_downmix || is_volume_used) {
31  ESP_LOGD("VolumeControl", "update_audio_data");
32  int32_t volumeFactor = get_volume_factor(volume);
33  int32_t max = get_volume_factor_max();
34  for (int i=0;i<frameCount;i++){
35  int32_t pcmLeft = data[i].channel1;
36  int32_t pcmRight = data[i].channel2;
37  // if mono -> we provide the same output on both channels
38  if (mono_downmix) {
39  pcmRight = pcmLeft = (pcmLeft + pcmRight) / 2;
40  }
41  // adjust the volume
42  if (is_volume_used) {
43  pcmLeft = pcmLeft * volumeFactor / max;
44  pcmRight = pcmRight * volumeFactor / max;
45  }
46  data[i].channel1 = pcmLeft;
47  data[i].channel2 = pcmRight;
48  }
49  }
50  }
51  // provides a factor in the range of 0 to 4096
52  virtual int32_t get_volume_factor(uint8_t volume) = 0;
53 
54  // provides the max factor value 4096
55  virtual int32_t get_volume_factor_max() {
56  return 0x1000;
57  }
58 };

65 class DefaultVolumeControl : public VolumeControl {
66  // provides a factor in the range of 0 to 4096
67  virtual int32_t get_volume_factor(uint8_t volume) {
68  constexpr double base = 1.4;
69  constexpr double bits = 12;
70  constexpr double zero_ofs = pow(base, -bits);
71  constexpr double scale = pow(2.0, bits);
72  double volumeFactorFloat = (pow(base, volume * bits / 127.0 - bits) - zero_ofs) * scale / (1.0 - zero_ofs);
73  int32_t volumeFactor = volumeFactorFloat;
74  if (volumeFactor > 0x1000) {
75  volumeFactor = 0x1000;
76  }
77  return volumeFactor;
78  }
79 
80 };

86 class SimpleExponentialVolumeControl : public VolumeControl {

88  virtual int32_t get_volume_factor(uint8_t volume) {
89  double volumeFactorFloat = volume;
90  volumeFactorFloat = pow(2.0, volumeFactorFloat * 12.0 / 127.0);
